<div class="container">
  <h2>استعلام عن نتيجة الامتحانات</h2>
  <input type="text" id="seatNumber" placeholder="أدخل رقم الجلوس...">
  <br>
  <button onclick="searchResult()">بحث</button>
  <div id="loading" class="loading">جاري جلب النتيجة من السيرفر...</div>
  <div id="error" class="error"></div>
  <div id="resultBox"></div>
</div>

<script>
  // ⚠️ استبدل الرابط اللي تحت ده برابط الـ API اللي نسخته من جوجل في الخطوة الأولى
  const API_URL = "https://script.google.com/macros/s/AKfycbx00I2C8ggJbaOsqN2PSzjrNBPw8jMePctm3heA62u1_bgWcow72IjHUhqeBHcJR5wZ/exec";

  async function searchResult() {
    var seatNum = document.getElementById("seatNumber").value.trim();
    var loading = document.getElementById("loading");
    var errorDiv = document.getElementById("error");
    var resultBox = document.getElementById("resultBox");
    
    if(!seatNum) { errorDiv.innerHTML = "برجاء كتابة رقم الجلوس"; return; }
    
    errorDiv.innerHTML = ""; resultBox.style.display = "none"; loading.style.display = "block";
    
    try {
      // استدعاء البيانات من جوجل شيت
      let response = await fetch(${API_URL}?id=${seatNum});
      let student = await response.json();
      
      loading.style.display = "none";
      if(student) {
        let html = "<h3>النتيجة:</h3>";
        for (let key in student) {
          html += <div class='result-item'><strong>${key}:</strong> ${student[key]}</div>;
        }
        resultBox.innerHTML = html;
        resultBox.style.display = "block";
      } else {
        errorDiv.innerHTML = "رقم الجلوس غير صحيح أو لم ترصد نتيجته بعد.";
      }
    } catch (err) {
      loading.style.display = "none";
      errorDiv.innerHTML = "حدث خطأ في الاتصال بالسيرفر، حاول مجدداً.";
      console.error(err);
    }
  }
</script>
